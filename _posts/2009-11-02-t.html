#include <stdio.h>
#include <errno.h>

int main(void)
{
FILE *fp;
char ch;

if(!(fp = fopen(__FILE__, "w+")))
{
fprintf(stderr, "%s\n", strerror(errno));
return(1);
}

fprintf(fp, "Hello World\n");

rewind(fp);
while((ch = getc(fp)) != EOF)
putchar(ch);

return(0);
}
