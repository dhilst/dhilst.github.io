#include <stdio.h>

struct alf
{ 
 char lt;
 struct alf *next;
}ar[27];


int main()
{
 struct alf *list_p;
 char ch, *str, *ar_p[12];
 short indx;
 ch = 97;
 str = "hello world";

 for(indx = 0; indx <= 25; indx++, ch++)
 {
  ar[indx].lt = ch;
  ar[indx].next = &ar[indx+1];
 }
 ar[26].lt = ' ';
 ar[26].next = (void *)0;


 for(indx = 0; *str != 0; str++)
 {
  list_p = ar; /* back 'a' */
  while(list_p)
  {
   if(list_p->lt == *str)
   ar_p[indx++] = &list_p->lt;
   list_p = list_p->next;
  }
 }
 *ar_p[11] = '\0'; 

 for(indx = 0; *ar_p[indx] != '\0'; indx++)
  printf("%c", *ar_p[indx]);
 putchar('\n'); 

 return(0);
}
