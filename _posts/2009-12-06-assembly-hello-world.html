section .data
 str: db  "Hello World",10
 strL: equ  $-str

section .text
 global _start:
_start:
 mov  eax, 4
 mov  ebx, 1
 mov  ecx, str
 mov  edx, strL
 int   0x80

 mov  eax, 1
 mov  ebx, 0
 int  0x80
