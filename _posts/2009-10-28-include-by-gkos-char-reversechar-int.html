char *reverse(char *p){
 char *q, *r, *s;
 void *malloc();
 q = p;
 r = s = (char *)malloc(sizeof p);
 for(;*p != 0; p++)
  ; /*_*/
 for(--p; p != q; p--, s++)
  *s = *p;
 *s = *p; *++s = 0;
 return(r);
}
