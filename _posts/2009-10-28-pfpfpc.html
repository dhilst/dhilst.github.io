#include <stdio.h>

char *ret_str(void){
return("Hello World");
}

char *(*(func)(void))(void){
char *(*pf)(void);
pf = ret_str;
return(pf);
}

int main(){
char *(*pff)(void);
pff = func();
printf("%s", pff());
return(0);
}

OUTPUT
Hello World
