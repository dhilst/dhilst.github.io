#include <stdio.h>
#include "bit.h"



int
main(int argc, char **argv)
{
 unsigned byte;
 argv++;
 for(; argc > 1; argc--, argv++)
 {
  byte = Bit2Byte(*argv);
  putchar((char)byte);
 }
 putchar('\n');
 return(0);
}

INPUT: hello-world-bin.o 01001000 01100101 01101100 01101100 01101111 00100000 01010111 01101111 01110010 01101100 01100100
OUTPUT: Hello World
